dropCircles

This module will be introduced via annotated code, which generates the image

Here's the code:

import {rs as circlePP} from '/shape/circle.mjs';
import {rs as basicP} from '/generators/basics.mjs';
import {rs as addDropMethods} from '/mlib/dropCircles.mjs';

let rs = basicP.instantiate();
addDropMethods(rs);

rs.setName('drop_light');
let ht= 6000;
let topParams = {width:ht,height:ht}
Object.assign(rs,topParams);

let dropParams = {dropTries:150}

rs.initProtos = function () {
  let circleP = this.circleP = circlePP.instantiate();
  circleP.fill = 'black';
  circleP['stroke-width'] = 0;
}  

rs.radiusGenerator= function (p) {
  let ln = p.length()-400;
  return 0.01*ln;
 }

rs.initialize = function () {
  this.initProtos();
  this.addRectangle({width:ht,height:ht,stroke_width:0,fill:'white'}); // defined in the basics module
  let shapes = this.set('shapes',arrayShape.mk());
  let drop =  this.generateCircleDrop(dropParams);
  let {points,radii} = drop;
  let ln  = points.length;
  for (let i=0;i<ln;i++) {
    let crc = this.circleP.instantiate();
    crc.dimension = 2*radii[i]; 
    shapes.push(crc);
    crc.moveto(points[i]);
   }
}

export {rs};

Like all the other generators, generateCircleDrop({dropTries:integer,radius:number}) will take its parameters from this, if not present in the params argument.

The circleDrop algorithm drops circles at random positions on the canvas. If a circle lands on top of another, it is thrown away, otherwise it is added. If defined, this.radiusGenerator(p:Point) is used to compute the radius of each circle dropped from the randomly chosen point p at which the drop attempt is made. If this.radiusGenerator returns a value that is less than or equal to zero, no drop attempt is made. If this.radiusGenerator is not defined, then the radius of each circle is the same, namely the value specified by the radius parameter passed to generateCircleDrop.

generateCircleDrop returns an object of the form {points:array of Point,radii:array of number}, specifying the successful drops. The algorithm terminates when dropTries successive attempts at a drop fail.
